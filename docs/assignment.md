#### 解构赋值
```
// 数值的解构赋值
[a, b, c] = [11, 22, 33]
a;    // 11
b;    // 22
c;    // 33

// 三个点的用法
// 对象的解构赋值
var {a, b, ...c} = {a: 123, b: 456, c: 789, d: 101112, e: 131415};
a;          // 123
b;          // 456
c;          // {c: 789, d: 101112, e: 131415}， c是对象
typeof c;   // "object"
d;          // 报错，d is not defined

// 数组的解构赋值
var [a, b, ...c] = [ 123, 456, 789,  101112, 131415]
a;          // 123
b;          // 456
c;          // [789, 101112, 131415], c是数组
d;          // 报错，d is not defined

// 交换变量
var x = 1, y = 2
[x, y] = [y, x]
x        // 2
y        // 1

// 下面是错误的写法：
// 错误：左边是数组，右边是对象
var [a, b, c] = {a: 123, b: 456, c: 789};
// 错误：左边是数组，右边是数组，但是右边的每个值是对象
var [a, b, c] = [a: 123, b: 456, c: 789]

```
